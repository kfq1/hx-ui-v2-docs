# Loading 加载

加载数据时显示动效。

# 全局-指令

showLoading
<template>
    <hx-button @click.native="showLoading">showLoading</hx-button>
</template>
<script>
    export default {
        name: "loading-demo",
        methods: {
            showLoading() {
                const loading = this.$loading({
                    lock: true,
                    text: 'Loading1'
                })
                setTimeout(() => {
                    loading.close()
                }, 2000)
            },
        }
    }
</script>
