# VirtualList 虚拟列

虚拟列表

# 基本用法

<template>
    <div class="virtual-list-demo" id="virtualListDemo">
        <virtual-list :data="list" v-slot="{item}">
            <div class="scroll-item" :style="`height:${item.height}px`" >
                {{item.index}}
            </div>
        </virtual-list>
    </div>
</template>
<script>
    export default {
        name: "list-demo",
        data() {
            return {
                list: []
            }
        },
        created() {
            for (let i = 0; i < 100; i++) {
                this.list[i] = {
                    index: i,
                    height: 40
                }
            }
        }
    }
</script>
<style>
.virtual-list-demo {
    height: 400px;
    margin: 0 auto;
    position: relative;
    text-align: center;
}
.scroll-item {
    padding: 3px 0;
    border-bottom: 1px solid red;
    box-sizing: border-box;
}
</style>
